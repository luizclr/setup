
# git aliases
alias ga="git add"
alias gb="git branch"
alias gcb="git checkout -b"
alias gc="git commit -m"
alias gcm="git checkout master"
alias gco="git checkout"
alias gd="git dif"
alias gds="git diff --staged"
alias gincoming="git !(git fetch --quiet && git log --pretty=format:'%C(yellow)%h %C(white)- %C(red)%an %C(white)- %C(cyan)%d%Creset %s %C(white)- %ar%Creset' ..@{u})"
alias glg="git log --pretty=format:'%Cred%h%Creset %C(bold)%cr%Creset %Cgreen<%an>%Creset %s' --max-count=30"
alias goutgoing="git !(git fetch --quiet && git log --pretty=format:'%C(yellow)%h %C(white)- %C(red)%an %C(white)- %C(cyan)%d%Creset %s %C(white)- %ar%Creset' @{u}..)"
alias gpl="git pull"
alias gpr="git pull --rebase"
alias gps="git push origin"
alias grollback="git reset --soft HEAD~1"
alias gs="git status -sb"
alias gundo="git checkout --"
alias gunstage="git reset HEAD --"
alias sf="git show --name-only"
